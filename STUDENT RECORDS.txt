#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Details {
    char name[20];
    char dept[10];
};

struct Student {
    int roll;
    struct Details info;
    struct Student *next;
};

int main() {
    struct Student *s1, *s2, *temp;

    s1 = (struct Student *)malloc(sizeof(struct Student));
    s2 = (struct Student *)malloc(sizeof(struct Student));

    s1->roll = 1;
    strcpy(s1->info.name, "Arun");
    strcpy(s1->info.dept, "CSE");
    s1->next = s2;

    s2->roll = 2;
    strcpy(s2->info.name, "Meena");
    strcpy(s2->info.dept, "IT");
    s2->next = NULL;

    temp = s1;
    while (temp != NULL) {
        printf("%d %s %s -> ",
               temp->roll,
               temp->info.name,
               temp->info.dept);
        temp = temp->next;
    }
    printf("NULL");

    return 0;
}
