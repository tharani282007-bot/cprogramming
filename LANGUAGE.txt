#include<stdio.h>
int main(){
    int lang;
    printf("1,tamil\n");
    printf("2,english\n");
    printf("3,telugu\n");
    printf("Enter the choice: ");
    scanf("%d",&lang);
    
   
   
    switch(lang){
        case 1:
        printf("You have choose tamil\n");
        break;
        case 2:
        printf("You have choose english\n");
        break;
        case 3:
        printf("You have choose telugu\n");
        break;
        default:
        printf("Invalid language");
        return 0;
        
    }
}
